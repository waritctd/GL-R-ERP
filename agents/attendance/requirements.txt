# only the Python standard library (ctypes) -- no pyzk.
#
# Run it with 32-bit Python on Windows (plcommpro.dll is 32-bit).
#
# Dependencies:
#   requests - posting punches to the ERP.
#   tzdata   - IANA timezone database. Windows ships no system tz data, so
#              zoneinfo.ZoneInfo("Asia/Bangkok") fails without this package.
requests>=2.31.0
tzdata>=2024.1
